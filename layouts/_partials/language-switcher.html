{{- $currentLang := .Language.Lang -}} {{- $currentLangName := .Language.LanguageName -}} {{- $isRegularPage := and (ne .Kind "home") (ne .Kind "section") (ne .Kind "taxonomy") (ne .Kind "term") -}}
{{- range .Site.Languages -}} {{- $lang := .Lang -}} {{- $isCurrent := eq $lang $currentLang -}} {{- $targetURL := "#" -}} {{- $hasTranslation := false -}} {{- if not $isCurrent -}} {{- /* Check if current page has a translation in this language */ -}} {{- if $.IsTranslated -}} {{- range $.Translations -}} {{- if eq .Language.Lang $lang -}} {{- $targetURL = .RelPermalink -}} {{- $hasTranslation = true -}} {{- end -}} {{- end -}} {{- end -}} {{- /* If no translation found, go to language homepage */ -}} {{- if not $hasTranslation -}} {{- $targetURL = printf "/%s/" $lang -}} {{- end -}} {{- end -}} {{- /* Show language if: it's current OR (it's a regular page AND has translation) OR (not a regular page - show all) */ -}} {{- $shouldShow := or $isCurrent (or (not $isRegularPage) $hasTranslation) -}} {{- if $shouldShow -}} {{ .LanguageName }} {{ if $isCurrent }}{{ end }} {{- end -}} {{- end -}}
